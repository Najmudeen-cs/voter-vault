Database Setup
Connect to your Supabase project
Supabase credentials
1. Go to supabase.com → create a free project
2. Dashboard → Settings → API
3. Copy Project URL and anon public key below
4. Run the SQL schema below in SQL Editor, then click Connect
SQL schema — run this in Supabase SQL Editor
-- Voters table
CREATE TABLE IF NOT EXISTS voters (
  id             BIGSERIAL PRIMARY KEY,
  voter_id       TEXT UNIQUE NOT NULL,
  full_name      TEXT,
  dob            TEXT,
  gender         TEXT,
  nationality    TEXT,
  aadhaar_id     TEXT,
  pan_id         TEXT,
  constituency   TEXT,
  address        TEXT,
  candidate_vote TEXT,
  registered_on  TIMESTAMPTZ,
  face_thumb     TEXT,
  created_at     TIMESTAMPTZ DEFAULT NOW()
);

-- Audit log table
CREATE TABLE IF NOT EXISTS audit_log (
  id         BIGSERIAL PRIMARY KEY,
  voter_id   TEXT,
  action     TEXT,
  result     TEXT,
  detail     TEXT,
  timestamp  TIMESTAMPTZ DEFAULT NOW()
);

-- Enable Row Level Security (allow anon read for admin panel)
ALTER TABLE voters    ENABLE ROW LEVEL SECURITY;
ALTER TABLE audit_log ENABLE ROW LEVEL SECURITY;

-- Policies: allow all for anon (tighten in production!)
CREATE POLICY "allow_all_voters"    ON voters    FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "allow_all_audit_log" ON audit_log FOR ALL USING (true) WITH CHECK (true);
